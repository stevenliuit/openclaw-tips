OpenAI Function Calling 完全指南：让 AI 调用真实工具

2026-03-21

Function Calling 是让大语言模型连接真实世界的桥梁。本文详细讲解如何定义函数、处理调用请求并实现完整的工具调用流程。

1. 函数定义

from openai import OpenAI

client = OpenAI(api_key="your-api-key")

# 定义可调用的函数
functions = [
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "获取指定城市的天气信息",
            "parameters": {
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称，如北京、上海"
                    },
                    "unit": {
                        "type": "string",
                        "enum": ["celsius", "fahrenheit"],
                        "description": "温度单位"
                    }
                },
                "required": ["city"]
            }
        }
    },
    {
        "type": "function", 
        "function": {
            "name": "calculate",
            "description": "执行数学计算",
            "parameters": {
                "type": "object",
                "properties": {
                    "expression": {
                        "type": "string",
                        "description": "数学表达式，如 2+3*4"
                    }
                },
                "required": ["expression"]
            }
        }
    }
]

2. 发送带函数调用的请求

# 第一次调用：让模型决定是否调用函数
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {
            "role": "user",
            "content": "北京今天天气怎么样？帮我算一下 123 * 456"
        }
    ],
    tools=functions,
    tool_choice="auto"  # 让模型自动选择
)

# 获取响应
message = response.choices[0].message
print(f"角色: {message.role}")
print(f"内容: {message.content}")
print(f"工具调用: {message.tool_calls}")

3. 处理工具调用

# 模拟工具函数实现
def execute_tool(tool_call):
    function_name = tool_call.function.name
    arguments = json.loads(tool_call.function.arguments)
    
    if function_name == "get_weather":
        # 实际项目中调用天气 API
        city = arguments["city"]
        unit = arguments.get("unit", "celsius")
        weather_data = {"temp": 25, "condition": "晴"}
        return json.dumps(weather_data)
    
    elif function_name == "calculate":
        expression = arguments["expression"]
        result = eval(expression)  # 实际使用安全的计算器
        return json.dumps({"result": result})
    
    return json.dumps({"error": "未知函数"})

# 执行所有工具调用
if message.tool_calls:
    # 添加助手消息（包含工具调用）
    messages = [
        {"role": "user", "content": "北京今天天气怎么样？帮我算一下 123 * 456"},
        message
    ]
    
    for tool_call in message.tool_calls:
        # 执行工具
        tool_result = execute_tool(tool_call)
        
        # 添加工具结果消息
        messages.append({
            "role": "tool",
            "tool_call_id": tool_call.id,
            "content": tool_result
        })
    
    # 第二次调用：获取最终回复
    final_response = client.chat.completions.create(
        model="gpt-4o",
        messages=messages,
        tools=functions
    )
    
    print(final_response.choices[0].message.content)

4. 强制调用特定函数

# 强制使用特定函数（不讓模型选择）
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "计算 100 + 200"}],
    tools=functions,
    tool_choice={
        "type": "function",
        "function": {"name": "calculate"}
    }
)

5. 完整对话示例

import json
from openai import OpenAI

client = OpenAI()

def chat_with_tools(user_message: str, functions: list) -> str:
    messages = [{"role": "user", "content": user_message}]
    
    # 第一轮：获取模型响应
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=messages,
        tools=functions
    )
    
    assistant_msg = response.choices[0].message
    messages.append(assistant_msg)
    
    # 如果有工具调用，执行并返回结果
    if assistant_msg.tool_calls:
        for tool_call in assistant_msg.tool_calls:
            # 执行工具
            result = run_tool(tool_call.function.name, 
                            json.loads(tool_call.function.arguments))
            
            # 添加结果
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": str(result)
            })
        
        # 第二轮：获取最终回复
        final_response = client.chat.completions.create(
            model="gpt-4o",
            messages=messages,
            tools=functions
        )
        return final_response.choices[0].message.content
    
    return assistant_msg.content

# 使用示例
result = chat_with_tools("上海天气如何？100的阶乘是多少？", functions)
print(result)

6. Structured Outputs（结构化输出）

GPT-4o 支持强制输出符合 schema 的结构化数据：

from pydantic import BaseModel
from typing import Literal

# 定义输出结构
class WeatherResponse(BaseModel):
    city: str
    temperature: float
    unit: Literal["celsius", "fahrenheit"]
    condition: str
    humidity: int

# 使用 structured outputs
response = client.chat.completions.create(
    model="gpt-4o-2024-08-06",
    messages=[{"role": "user", "content": "描述北京天气"}],
    response_format={"type": "json_object", "schema": WeatherResponse.model_json_schema()}
)

result = WeatherResponse.model_validate_json(response.choices[0].message.content)
print(f"{result.city}: {result.temperature}°{result.unit}, {result.condition}")
