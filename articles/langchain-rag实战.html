LangChain RAG 实战：构建企业级知识库问答系统

2026-03-21

本文详细介绍如何使用 LangChain 构建生产级 RAG（检索增强生成）系统，包含文档加载、向量存储、检索优化和答案生成的完整流程。

1. 环境准备

pip install langchain langchain-openai langchain-community \
    chromadb pypdf python-docx pydantic-settings

# 配置环境变量
export OPENAI_API_KEY="your-api-key"

2. 文档加载与处理

from langchain_community.document_loaders import (
    PyPDFLoader, DirectoryLoader, TextLoader
)
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings

# 支持多种文档格式
loaders = {
    "pdf": PyPDFLoader("docs/manual.pdf"),
    "txt": TextLoader("docs/readme.txt"),
    "docx": DirectoryLoader("docs/", glob="*.docx")
}

# 加载文档
documents = []
for doc_type, loader in loaders.items():
    documents.extend(loader.load())

# 文本分块
splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,
    chunk_overlap=200,
    separators=["\n\n", "\n", "。", " ", ""]
)

chunks = splitter.split_documents(documents)
print(f"生成 {len(chunks)} 个文本块")

3. 向量存储

from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

# 使用 OpenAI Embeddings
embeddings = OpenAIEmbeddings(
    model="text-embedding-3-small"
)

# 创建向量数据库
vectorstore = Chroma.from_documents(
    documents=chunks,
    embedding=embeddings,
    collection_name="company_knowledge",
    persist_directory="./chroma_db"
)

# 保存到磁盘
vectorstore.persist()
print("向量数据库已保存")

4. 构建 RAG 链

from langchain_openai import ChatOpenAI
from langchain import hub
from langchain_core.runnables import RunnablePassthrough

# 加载预定义的 RAG prompt
prompt = hub.pull("rlm/rag-prompt")

# 初始化 LLM
llm = ChatOpenAI(
    model="gpt-4o",
    temperature=0
)

# 创建检索器
retriever = vectorstore.as_retriever(
    search_type="similarity_score_threshold",
    search_kwargs={
        "k": 5,
        "score_threshold": 0.7
    }
)

# 构建 RAG 链
def format_docs(docs):
    return "\n\n".join(doc.page_content for doc in docs)

rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | prompt
    | llm
)

5. 高级检索策略

5.1 混合搜索

# 混合搜索：结合向量相似度和关键词匹配
from langchain_community.retrievers import BM25Retriever

# 关键词检索
keyword_retriever = BM25Retriever.from_documents(chunks)

# 向量检索
vector_retriever = vectorstore.as_retriever(search_kwargs={"k": 5})

# 组合检索器
from langchain.retrievers import EnsembleRetriever

ensemble_retriever = EnsembleRetriever(
    retrievers=[keyword_retriever, vector_retriever],
    weights=[0.3, 0.7]  # 关键词30%，向量70%
)

5.2 查询改写

from langchain.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

# 查询改写模板
rewrite_template = """将用户问题改写为更利于检索的版本：
原问题：{question}
改写后："""

rewrite_prompt = ChatPromptTemplate.from_template(rewrite_template)
rewrite_chain = rewrite_prompt | llm

# 使用改写后的查询
def retrieve_with_rewrite(question):
    # 改写问题
    rewritten = rewrite_chain.invoke({"question": question}).content
    # 使用改写后的问题检索
    docs = ensemble_retriever.get_relevant_documents(rewritten)
    return docs

6. 完整示例

# 问答系统完整实现
class RAGQASystem:
    def __init__(self, vectorstore_path: str):
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            persist_directory=vectorstore_path,
            embedding_function=self.embeddings
        )
        self.llm = ChatOpenAI(model="gpt-4o")
        self.retriever = self.vectorstore.as_retriever(k=5)
        self.chain = self._build_chain()
    
    def _build_chain(self):
        prompt = hub.pull("rlm/rag-prompt")
        return (
            {"context": self.retriever | format_docs, "question": RunnablePassthrough()}
            | prompt
            | self.llm
        )
    
    def ask(self, question: str) -> str:
        response = self.chain.invoke(question)
        return response.content

# 使用
qa_system = RAGQASystem("./chroma_db")
answer = qa_system.ask("公司的年假政策是什么？")
print(answer)

7. 生产部署建议

  • 缓存优化：使用 Redis 缓存频繁查询的结果
  • 异步处理：使用 asyncio 提高并发处理能力
  • 监控告警：记录检索质量指标（召回率、相关性）
  • 增量更新：支持文档的增删改，无需重建整个向量库

总结

LangChain 提供了完整的 RAG 开发框架。通过合理的检索策略和优化，可以构建高质量的企业知识库问答系统。
